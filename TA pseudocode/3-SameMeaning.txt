=> INPUT KATA // User input KODOK
=> PANJANG_KATA = LENG KATA

=> kata_awal = PANJANG_KATA
=> kata_akhir = 0

=> output = TRUE 

=> WHILE kata_awal = PANJANG_KATA

	-END IF

	-kata_akhir ++

=> DISPLAY output
=> DISPLAY "SELESAI"
